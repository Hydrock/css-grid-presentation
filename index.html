                grid-column-start: 2;

                grid-row-start: row1-start
                grid-row-end: 3
              }

Вот так вот:

            
              .item-b {
                grid-column-start: 1;
                grid-column-end: span col4-start;
                grid-row-start: 2
                grid-row-end: span 2
              }
